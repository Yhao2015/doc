    # jw-table 表格

    # 基础用法

    jw-table 图表组件示例，请注意：数据格式需要按照规范格式进行传入。

    <template>
    <div>
        <jw-table
            skin="light"
            :columns="columns" 
            :tableData="sourceData" 
            :pagination="true"
            :loading="loading"
            :pageNum="pageIndex" 
            :total="total" 
            :pageSize="pageSize"   
            class="marginT24"
            @changePage="onChangePagination" 
        >
            <template slot="suggestion" slot-scope="{ data }">
                <a @click="onClick(data.record)">查看</a>
            </template>
        </jw-table>
    </div>
</template>

<script>
const columns = [
    {
        title: '告警来源',
        dataIndex: 'alarmSource',
        ellipsis: true,
        align: 'center'
    },
    {
        title: '告警等级',
        dataIndex: 'alarmLevel',
        align: 'center'
    },
    {
        title: '告警类型',
        dataIndex: 'alarmType',
        align: 'center'
    },
    {
        title: '告警位置',
        dataIndex: 'dceName',
        align: 'center'
    },
    {
        title: '告警内容',
        dataIndex: 'content',
        align: 'center'
    },
    {
        title: '告警原因',
        dataIndex: 'reason',
        align: 'center'
    },
    {
        title: '发生时间',
        dataIndex: 'alarmTime',
        align: 'center'
    },
    {
        title: '告警状态',
        dataIndex: 'alarmStatus',
        align: 'center'
    },
    {
        title: '告警详情及建议',
        dataIndex: 'suggestion',
        scopedSlots: { customRender: 'suggestion' },
        align: 'center'
    }
]
export default {
    data() {
        return {
            columns,
            loading: false,
            pageSize: 15,
            pageIndex: 1,
            total: 1,
            sourceData: [
                {
                    alarmSource: 'abc',
                    alarmLevel: '危险'
                }
            ]
        }
    },
    methods: {
        getTableData() {
            this.sourceData = []
        },
        onChangePagination(page) {
            this.pageIndex = page
            this.getTableData()
        }
    }
}
</script>
