    # jw-form 表单

    # 基础用法

    jw-form 组件示例，请注意：数据格式需要按照规范格式进行传入。

    <template>
    <div>
        <jw-form 
            :formConfigData="formConfigData" 
            :formConfigBase="formConfigBase"
            :formFunctionList="functions"
            theme="light" 
            ref="search"
        >
            <a-col slot="formExpand" class="marginT4">
                <a-space class="lineHeight36 marginL8 pull-right" :size="16">
                    <a-button type="primary" ghost icon="search"> 查询 </a-button>
                    <a-button type="primary" ghost icon="plus"> 新增 </a-button>
                </a-space>
            </a-col>
        </jw-form>
    </div>
</template>

<script>
const formConfigData = [
    {
        label: '中文名称',
        code: 'name',
        type: 'input',
        show: true
    },
    {
        label: '手机号码',
        code: 'username',
        type: 'input',
        show: true,
        changeFunName: "onChange",
        blurFunName: "onBlur",
        focusFunName: "onFocus",
    }
]
const formConfigBase = {
    labelCol: null,
    wrapperCol: null,
    formName: 'searchForm',
    layout: 'inline',
    labelAlign: 'left',
    grid: {
        xxs: { span: 6 },
        xs: { span: 6 },
        sm: { span: 6 },
        md: { span: 6 },
        lg: { span: 6 },
        xl: { span: 6 },
        xxl: { span: 6 }
    }
}
export default {
    data() {
        return {
            formConfigData,
            formConfigBase,
            functions: {
                onChange: this.onChange,
                onBlur: this.onBlur,
                onFocus: this.onFocus,
            }
        }
    },
    methods: {
        onChange() {

        },
        onBlur() {

        },
        onFocus() {
            
        }
    }
}
</script>
